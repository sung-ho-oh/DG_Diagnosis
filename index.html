📊 Google Sheets 연동

Apps Script 코드 (새 시트에 붙여넣기):
function doPost(e) { try { if (!e || !e.postData) return ContentService.createTextOutput('no data'); // form submit 방식: e.parameter.payload 로 수신 var d = JSON.parse(e.parameter.payload); var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); if (sheet.getLastRow() === 0) { sheet.appendRow(['응답일시','기수','이름','소속/직급','사전/사후', 'AI경력','사용도구','Model','Context','Structure','합계','AI피드백']); } sheet.appendRow([new Date(),d.cohort,d.name,d.org,d.prepost, d.aiExp,d.tools,d.M,d.C,d.S,d.total,d.feedback]); return ContentService.createTextOutput('ok'); } catch(err) { return ContentService.createTextOutput('error:'+err.toString()); } }
